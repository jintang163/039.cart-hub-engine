组件使用方式

// 1. 初始化 SDK
const sdk = CartHubSDK.create({
    baseUrl: 'http://localhost:8080',
    tenantId: 'default',
    bizType: 'ecommerce',
    userId: 'user_1001',
    debug: true
});

// 2. 获取过期信息
const expireInfo = await sdk.getExpireInfo();

// 3. 或者在购物车数据中直接读取
const cart = await sdk.getCart();
console.log(cart.isExpiring, cart.daysLeft, cart.expireTime);

// 4. 监听事件
sdk.on('expireInfoLoaded', (info) => {
    if (info.isExpiring) {
        renderExpireBanner(info);
    }
});
